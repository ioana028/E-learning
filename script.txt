DROP TABLE USERS CASCADE CONSTRAINTS;
DROP TABLE CHAPTERS CASCADE CONSTRAINTS;
DROP TABLE LESSONS CASCADE CONSTRAINTS;
DROP TABLE EXERCISES CASCADE CONSTRAINTS;
DROP TABLE USER_ERRORS CASCADE CONSTRAINTS;


CREATE TABLE users (
  user_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  username VARCHAR2(50) UNIQUE NOT NULL,
  password VARCHAR2(255) NOT NULL,
  email VARCHAR2(100),
  role VARCHAR2(20) DEFAULT 'student', -- rolul utilizatorului: student/admin/profesor
  last_completed_chapter NUMBER(2) DEFAULT 0,
  LAST_COMPLETED_LESSONS CLOB
  english_level VARCHAR2(10);
);

CREATE TABLE CHAPTERS (
    ID NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    TITLE VARCHAR2(255) NOT NULL,
    DESCRIPTION VARCHAR2(500)
);

CREATE TABLE LESSONS (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    CHAPTER_ID NUMBER NOT NULL,
    TITLE VARCHAR2(255) NOT NULL,
    LESSON_NUMBER NUMBER(2)
);

ALTER TABLE LESSONS ADD CONSTRAINT FK_LESSON_CHAPTER FOREIGN KEY (CHAPTER_ID) REFERENCES CHAPTERS(ID) ON DELETE CASCADE;


CREATE TABLE EXERCISES(
    ID NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    TYPE NUMBER(2),
    LESSON_ID NUMBER,
    QUESTION VARCHAR2(200),
    OPTIONS VARCHAR2(200),
    ANSWER VARCHAR2(100),
    difficulty VARCHAR2(10), -- usor, mediu, greu
    topic VARCHAR2(50)       -- past tense, adjectives, infinitives etc.
    
);
ALTER TABLE EXERCISES ADD CONSTRAINT FK_EXERCISE_LESSON FOREIGN KEY (LESSON_ID) REFERENCES LESSONS(ID) ON DELETE CASCADE;

CREATE TABLE user_errors (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  user_id NUMBER NOT NULL,
  exercise_id NUMBER NOT NULL,
  topic VARCHAR2(50),
  difficulty VARCHAR2(10),
  error_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
